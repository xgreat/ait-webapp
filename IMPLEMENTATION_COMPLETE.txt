💻 USAGE EXAMPLE

import { getPosts } from '@/lib/wordpress';

export default async function BlogPage() {
  const posts = await getPosts({ per_page: 10 });
  
  return (
    <div>
      {posts.map(post => (
        <article key={post.id}>
          <h2>{post.title.rendered}</h2>
          <p>{post.excerpt.rendered}</p>
        </article>
      ))}
    </div>
  );
}
